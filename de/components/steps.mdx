import IconsOptional from "/snippets/icons-optional.mdx";

Verwenden Sie Schritte, um eine Reihe aufeinanderfolgender Aktionen oder Ereignisse darzustellen. Sie können beliebig viele Schritte hinzufügen.

<Steps>
  <Step title="Erster Schritt">
    Dies sind Anleitungen oder Inhalte, die nur für den ersten Schritt gelten.
  </Step>
  <Step title="Zweiter Schritt">
    Dies sind Anleitungen oder Inhalte, die nur für den zweiten Schritt gelten.
  </Step>
  <Step title="Dritter Schritt">
    Dies sind Anleitungen oder Inhalte, die nur für den dritten Schritt gelten.
  </Step>
</Steps>

```mdx Steps example
<Steps>
  <Step title="First Step">
    These are instructions or content that only pertain to the first step.
  </Step>
  <Step title="Second Step">
    These are instructions or content that only pertain to the second step.
  </Step>
  <Step title="Third Step">
    These are instructions or content that only pertain to the third step.
  </Step>
</Steps>
```

<div id="steps-properties">
  ## Eigenschaften von Steps
</div>

<ResponseField name="children" type="ReactElement<StepProps>[]" required>
  Eine Liste von `Step`-Komponenten.
</ResponseField>

<ResponseField name="titleSize" type="string" default="p">
  Die Größe der Schrittüberschriften. Einer von `p`, `h2` oder `h3`.
</ResponseField>

<div id="individual-step-properties">
  ## Eigenschaften einzelner Schritte
</div>

<ResponseField name="children" type="string | ReactNode" required>
  Der Inhalt eines Schritts, entweder als Klartext oder als Komponenten.
</ResponseField>

<IconsOptional />

<ResponseField name="title" type="string">
  Der Titel ist der Haupttext des Schritts und erscheint neben dem Indikator.
</ResponseField>

<ResponseField name="stepNumber" type="number">
  Die Nummer des Schritts.
</ResponseField>

<ResponseField name="titleSize" type="string" default="p">
  Die Größe der Schrittitel. Entweder `p`, `h2` oder `h3`.
</ResponseField>
